using System;
using System.Text;
using System.Net;
using System.Net.Sockets;
using System.IO;

namespace BinaryStreamApp
{
    class Program
	{
		public static void Main(string[] args)
		{
			FileStream fs;
			fs = new FileStream(@"C:\Users\mh.guo\Desktop\3.dat",FileMode.OpenOrCreate,FileAccess.ReadWrite);
			BinaryWriter bw = new BinaryWriter(fs);
			
			double aDouble = 1234.67;
			int anInt = 34567;
			char[] aCharArray = {'A','B','c','d'};
			bw.Write(aDouble);
			bw.Write(anInt);
			bw.Write(aCharArray);
			int length = Convert.ToInt32(bw.BaseStream.Length);
			fs.Close();
			bw.Close();
			
			fs = new FileStream(@"C:\Users\mh.guo\Desktop\3.dat",FileMode.OpenOrCreate,FileAccess.Read);
			BinaryReader br = new BinaryReader(fs);
			Console.WriteLine(br.ReadDouble().ToString());
			Console.WriteLine(br.ReadInt32().ToString());
			char[] data = br.ReadChars(length);
			for(int i = 0;i < data.Length;i++)
			{
				//x表示十六进制
			    Console.Write("{0,7:x}",data[i]);
			}
			fs.Close();
			br.Close();
			Console.ReadKey();
		}
	}
}
